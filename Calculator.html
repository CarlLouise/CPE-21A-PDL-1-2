def calculator():
    while True:
        operation = input("Enter an operation (+, -, *, /): ")
        if operation not in ["+", "-", "*", "/"]:
            print("Invalid operation. Please try again.")
            continue
        num1 = float(input("Enter the first number: "))
        num2 = float(input("Enter the second number: "))
        if operation == "+":
            print("The result is: ", num1 + num2)
        elif operation == "-":
            print("The result is: ", num1 - num2)
        elif operation == "*":
            print("The result is: ", num1 * num2)
        elif operation == "/":
            print("The result is: ", num1 / num2)

calculator()
